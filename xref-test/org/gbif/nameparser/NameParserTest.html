111   private ParsedName buildParsedNameFromTabRow(String[] cols) {
112     ParsedName pn = null;
113     try {
114       pn = new ParsedName(NameType.fromString(cols[13]), StringUtils.trimToNull(cols[1]),
115           StringUtils.trimToNull(cols[2]), StringUtils.trimToNull(cols[3]), StringUtils.trimToNull(cols[4]),
116           NamePart.fromString(cols[12]), StringUtils.trimToNull(cols[10]), StringUtils.trimToNull(cols[6]),
117           StringUtils.trimToNull(cols[7]), StringUtils.trimToNull(cols[8]), StringUtils.trimToNull(cols[9]),
118           StringUtils.trimToNull(cols[5]), null, null, null, null);
119     } catch (ArrayIndexOutOfBoundsException e) {
120       LOG.error("scientific_names.txt file bogus, too little columns:\n{}", Joiner.on("|").useForNull("").join(cols));
121       throw e;
122     }
123     return pn;
124   }

842   @Test
843   public void testNameFile() throws Exception {
844     final int CURRENTLY_FAIL = 24;
845     final int CURRENTLY_FAIL_EXCL_AUTHORS = 19;
846 
847     LOG.info("\n\nSTARTING FULL PARSER\n");
848     Reader reader = FileUtils.getInputStreamReader(streamUtils.classpathStream("scientific_names.txt"));
849     LineIterator iter = new LineIterator(reader);
850 
851     int parseFails = 0;
852     int parseNoAuthors = 0;
853     int parseErrors = 0;
854     int cnt = 0;
855     int lineNum = 0;
856     long start = System.currentTimeMillis();
857 
858     Splitter tabSplit = Splitter.on('\t');
859     while (iter.hasNext()) {
860       lineNum++;
861       String line = iter.nextLine();
862       if (lineNum == 1 || line == null || line.startsWith("#") || line.trim().isEmpty()) {
863         continue;
864       }
865       cnt++;
866 
867       List<String> cols = tabSplit.splitToList(line);
868       String name = cols.get(0);
869       if (cols.size() < 14) {
870         LOG.warn("Too short line in names test file:");
871         LOG.warn(line);
872       }
873       ParsedName expected = buildParsedNameFromTabRow(cols.toArray(new String[cols.size()]));
874       ParsedName n;
875       try {
876         n = parser.parse(name, null);
877         if (!n.isAuthorsParsed()) {
878           parseNoAuthors++;
879         }
880         // copy scientific name, we dont wanna compare it, it might be slightly different
881         expected.setScientificName(n.getScientificName());
882         // remove SCIENTIFIC nametype as we only like to compare other values
883         if (NameType.SCIENTIFIC == n.getType()) {
884           n.setType(null);
885         }
886         if (!n.equals(expected)) {
887           parseErrors++;
888           LOG.warn("WRONG\t " + name + "\n  EXPECTED: " + expected + "\n  PARSED  : " + n);
889         }
890       } catch (Exception e) {
891         parseFails++;
892         LOG.warn("FAIL\t " + name + "\nEXPECTED: " + expected);
893       }
894     }
895     long end = System.currentTimeMillis();
896     int successfulParses = cnt - parseFails - parseErrors;
897     LOG.info("\n\nNames tested: " + cnt);
898     LOG.info("Names success: " + successfulParses);
899     LOG.info("Names parse fail: " + parseFails);
900     LOG.info("Names parse errors: " + parseErrors);
901     LOG.info("Names ignoring authors: " + parseNoAuthors);
902     LOG.info("Total time: " + (end - start));
903     LOG.info("Average per name: " + (((double) end - start) / cnt));
904 
905     if ((parseFails + parseErrors) > CURRENTLY_FAIL) {
906       TestCase.fail(
907           "We are getting worse, not better. Currently failing: " + (parseFails + parseErrors) + ". We used to fail :"
908               + CURRENTLY_FAIL);
909     } else if (parseFails + parseErrors < CURRENTLY_FAIL) {
910       LOG.info("We are getting better! Used to fail : " + CURRENTLY_FAIL);
911     }
912     if ((parseFails + parseNoAuthors) > CURRENTLY_FAIL_EXCL_AUTHORS) {
913       TestCase.fail(
914           "We are getting worse, not better. Currently failing ignroing authors: " + (parseFails + parseNoAuthors)
915               + ". Was passing:" + CURRENTLY_FAIL_EXCL_AUTHORS);
916     } else if (parseFails + parseNoAuthors < CURRENTLY_FAIL_EXCL_AUTHORS) {
917       LOG.info("We are getting better! Used to fail without authors: " + CURRENTLY_FAIL_EXCL_AUTHORS);
