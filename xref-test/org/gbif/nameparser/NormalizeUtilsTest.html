45    @Test
46    public void testReplaceUnicodeEntities() {
47      assertEquals("Markus&Pia", NormalizeUtils.replaceUnicodeEntities("Markus&Pia"));
48      assertEquals("Markus&Pia;", NormalizeUtils.replaceUnicodeEntities("Markus&Pia;"));
49      assertEquals("Markus & Pia ; ", NormalizeUtils.replaceUnicodeEntities("Markus & Pia ; "));
50      assertEquals("&#pia;", NormalizeUtils.replaceUnicodeEntities("&#pia;"));
51      assertEquals("&#12pia;", NormalizeUtils.replaceUnicodeEntities("&#12pia;"));
52      Assert.assertTrue(NormalizeUtils.replaceUnicodeEntities(null) == null);
53      assertEquals("лобан", NormalizeUtils.replaceUnicodeEntities("&#1083;&#1086;&#1073;&#1072;&#1085;"));
54      assertEquals("лобан", NormalizeUtils.replaceUnicodeEntities("&#x43b;&#x43e;&#x431;&#x430;&#x43d;"));
55      assertEquals("лобан", NormalizeUtils.replaceUnicodeEntities("&#x43B;&#x043e;&#x0431;&#x430;&#x43D;"));
56      //    assertEquals("лобан", NormalizeUtils.replaceUnicodeEntities("\\u043b\\u0u43e\\u0431\\u0430\\u043d"));
57      //    assertEquals("лобан", NormalizeUtils.replaceUnicodeEntities("\\u043B\\u0u43E\\u0431\\u0430\\u043D"));
58      //    assertEquals("\\u43Bобан", NormalizeUtils.replaceUnicodeEntities("\\u43B\\0u43E\\u0431\\u0430\\u043D"));
59      assertEquals("\\u43b\\u43e\\u431\\u430\\u43d",
