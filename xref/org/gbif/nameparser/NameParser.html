219   private static String cutoffBadSuffices(String name) {
220     boolean done = false;
221     String lowercase = name.toLowerCase();
222     int cuttoff = lowercase.length();
223     while (!done) {
224       done = true;
225       for (String bad : badNameParts) {
226         if (lowercase.endsWith(" " + bad)) {
227           int remove = bad.length() + 1;
228           cuttoff -= remove;
229           lowercase = lowercase.substring(0, cuttoff);
230           done = false;
231         }
232       }
233     }
234     if (cuttoff != name.length()) {
235       name = name.substring(0, cuttoff);
236     }
237     return name;
238   }

281   /**
282    * Carefully normalizes a scientific name trying to maintain the original as close as possible.
283    * In particular the string is normalized by:
284    * - adding commas in front of years
285    * - trims whitespace around hyphens
286    * - unescapes unicode chars \\uhhhh, \\nnn, \xhh
287    * - pads whitespace around &
288    * - adds whitespace after dots following a genus abbreviation or rank marker
289    * - keeps whitespace before opening and after closing brackets
290    * - removes whitespace inside brackets
291    * - removes whitespace before commas
292    * - normalized hybrid marker to be the ascii multiplication sign
293    * - removes whitespace between hybrid marker and following name part in case it is NOT a hybrid formula
294    * - trims the string and replaces multi whitespace with single space
295    * - capitalizes all only uppercase words (authors are often found in upper case only)
296    *
297    * @param name To normalize
298    *
299    * @return The normalized name
300    */
301   public static String normalize(String name) {
302     if (name == null) {
303       return null;
304     }
305     name = org.gbif.utils.text.StringUtils.unescapeUnicodeChars(name);
306 
307     // normalise usage of dots after abbreviated genus and rank marker
308     Matcher m = NORM_DOTS.matcher(name);
309     if (m.find()) {
310       name = m.replaceAll("$1. ");
311     }
312     // use commas before years
313     // ICZN §22A.2 http://www.iczn.org/iczn/includes/page.jsp?article=22&nfv=
314     m = COMMA_BEFORE_YEAR.matcher(name);
315     if (m.find()) {
316       name = m.replaceAll("$1, $2");
317     }
318     // no whitespace around hyphens
319     name = NORM_HYPHENS.matcher(name).replaceAll("-");
320     // use whitespace with &
321     name = NORM_AMPERSAND_WS.matcher(name).replaceAll(" & ");
322 
323     // whitespace before and after brackets, keeping the bracket style
324     m = NORM_BRACKETS_OPEN.matcher(name);
325     if (m.find()) {
326       name = m.replaceAll(" $1");
327     }
328     m = NORM_BRACKETS_CLOSE.matcher(name);
329     if (m.find()) {
330       name = m.replaceAll("$1 ");
331     }
332     // remove whitespace before commas and replace double commas with one
333     m = NORM_COMMAS.matcher(name);
334     if (m.find()) {
335       name = m.replaceAll(", ");
336     }
337     // normalize hybrid markers
338     m = NORM_HYBRIDS_GENUS.matcher(name);
339     if (m.find()) {
340       name = m.replaceFirst("×$1");
341     }
342     m = NORM_HYBRIDS_EPITH.matcher(name);
343     if (m.find()) {
344       name = m.replaceFirst("$1 ×$2");
345     }
346     m = NORM_HYBRIDS_FORM.matcher(name);
347     if (m.find()) {
348       name = m.replaceAll(" × ");
349     }
350     // capitalize all entire upper case words
351     m = NORM_UPPERCASE_WORDS.matcher(name);
352     while (m.find()) {
353       name = name.replaceFirst(m.group(0), m.group(1) + m.group(2).toLowerCase());
354     }
355 
356     // finally whitespace and trimming
357     name = NORM_WHITESPACE.matcher(name).replaceAll(" ");
358     return StringUtils.trimToEmpty(name);
359   }

361   /**
362    * Does the same as a normalize and additionally removes all ( ) and "und" etc
363    * Checks if a name starts with a blacklisted name part like "Undetermined" or "Uncertain" and only returns the
364    * blacklisted word in that case
365    * so its easy to catch names with blacklisted name parts.
366    *
367    * @param name To normalize
368    *
369    * @return The normalized name
370    */
371   protected static String normalizeStrong(String name) {
372     if (name == null) {
373       return null;
374     }
375     // normalize all quotes to single "
376     name = NORM_QUOTES.matcher(name).replaceAll("'");
377     // enclosing quotes
378     name = REPLACE_QUOTES.matcher(name).replaceAll("");
379     // no question marks after words (after years they should remain!)
380     Matcher m = NO_Q_MARKS.matcher(name);
381     if (m.find()) {
382       name = m.replaceAll("$1");
383     }
384 
385     // remove prefixes
386     name = NORM_PREFIXES.matcher(name).replaceAll("");
387 
388     // remove brackets inside the genus, the kind taxon finder produces
389     m = NORM_TF_GENUS.matcher(name);
390     if (m.find()) {
391       name = m.replaceAll("$1$2 ");
392     }
393     // replace imprint years. See ICZN §22A.2.3 http://www.iczn.org/iczn/index.jsp?nfv=&article=22
394     // Ctenotus alacer Storr, 1970 ["1969"] -> Ctenotus alacer Storr, 1970
395     // C. Flahault 1887 ("1886-1888") -> C. Flahault 1887
396     m = NORM_IMPRINT_YEAR.matcher(name);
397     if (m.find()) {
398       // System.out.println(m.group(0));
399       // System.out.println(m.group(1));
400       name = m.replaceAll("$1");
401     }
402 
403     // replace bibliographic in authorship
404     /*
405      * m = NORM_IN_BIB.matcher(name);
406      * if (m.find()) {
407      * // keep year if it only exists in IN reference
408      * Matcher mIN = EXTRACT_YEAR.matcher(m.group(0));
409      * name = m.replaceAll("");
410      * Matcher mNAME = EXTRACT_YEAR.matcher(name);
411      * if (mIN.find() && !mNAME.find()) {
412      * name = name + ", " + mIN.group(1);
413      * }
414      * }
415      */
416 
417     // This is redundant, as it is done in the regular normalize function already
418     // BUT somehow all upper case authors slow down parsing so much that it can even come to an hold in the next step
419     // so we pay the price and do it twice
420     // capitalize all entire upper case words
421     m = NORM_UPPERCASE_WORDS.matcher(name);
422     while (m.find()) {
423       name = name.replaceFirst(m.group(0), m.group(1) + m.group(2).toLowerCase());
424     }
425 
426     /*
427      * ICBN §46.2, Note 1.
428      * When authorship of a name differs from authorship of the publication in which it was validly published, both are
429      * sometimes cited,
430      * connected by the word "in". In such a case, "in" and what follows are part of a bibliographic citation and are
431      * better omitted
432      * unless the place of publication is being cited.
433      */
434 
435     // normalise original name authorship, putting author AND year in brackets
436     // with long authorships this gets slow. Test if year exists first:
437     m = EXTRACT_YEAR.matcher(name);
438     if (m.find() && name.length() < 80) {
439       m = NORM_ORIG_AUTH.matcher(name);
440       if (m.find()) {
441         name = m.replaceAll("($1 $2)");
442       }
443       m = NORM_ORIG_AUTH2.matcher(name);
444       if (m.find()) {
445         name = m.replaceAll("($1 $2)");
446       }
447     }
448 
449     // replace square brackets, keeping content (or better remove all within?)
450     name = NORM_NO_SQUARE_BRACKETS.matcher(name).replaceAll(" $1 ");
451     // replace different kind of brackets with ()
452     name = NORM_BRACKETS_OPEN_STRONG.matcher(name).replaceAll(" (");
453     name = NORM_BRACKETS_CLOSE_STRONG.matcher(name).replaceAll(") ");
454     // normalise different usages of ampersand, and, et &amp;
455     name = NORM_AND.matcher(name).replaceAll(" & ");
456     // but keep "et al." instead of "& al."
457     name = NORM_ET_AL.matcher(name).replaceAll(" et al.");
458 
459     // // add commans between authors in space delimited list
460     // m = NORM_AUTH_DELIMIT.matcher(name);
461     // if (m.find()){
462     // name = m.replaceAll("$1, $2");
463     // }
464     // Bryozoan indet. 1
465     // Bryozoa sp. 2
466     // Bryozoa sp. E
467 
468     name = NORM_SUFFIXES.matcher(name).replaceAll("");
469 
470     // add parenthesis around subgenus if missing
471     m = NORM_SUBGENUS.matcher(name);
472     if (m.find()) {
473       name = m.replaceAll("$1 ($2) $3");
474     }
475 
476     // finally whitespace and trimming
477     name = normalize(name);
478     // name = NORM_WHITESPACE.matcher(name).replaceAll(" ");
479     return StringUtils.trimToEmpty(name);

482   /**
483    * basic careful cleaning, trying to preserve all parsable name parts
484    */
485   protected static String preClean(String name) {
486     // unescape unicode
487     name = org.gbif.utils.text.StringUtils.unescapeUnicodeChars(name);
488     // TODO: unescape html entities
489     // replace xml tags
490     name = XML_TAGS.matcher(name).replaceAll("");
491     // trim
492     name = name.trim();
493     // remove quotes in beginning and matching ones at the end
494     for (char c : QUOTES) {
495       int idx = 0;
496       while (idx < name.length() && (c == name.charAt(idx) || Character.isWhitespace(name.charAt(idx)))) {
497         idx++;
498       }
499       if (idx > 0) {
500         // check if we also find this char at the end
501         int end = 0;
502         while (c == name.charAt(name.length() - 1 - end) && (name.length() - idx - end) > 0) {
503           end++;
504         }
505         name = name.substring(idx, name.length() - end);
506       }
507     }
508     name = NORM_WHITESPACE.matcher(name).replaceAll(" ");
509     return StringUtils.trimToEmpty(name);
510   }

526   private void checkEpithetVsAuthorPrefx(ParsedName cn) {
527     if (cn.getRankMarker() == null) {

530         // or short authorship prefix in epithet. test

538         }
539       } else {
540         // might be monomial with the author prefix erroneously taken as the species epithet
541         String extendedAuthor = cn.getSpecificEpithet() + " " + cn.getAuthorship();
542         Matcher m = AUTHOR_TEAM_PATTERN.matcher(extendedAuthor);
543         if (m.find()) {
544           // matches author. Prefer that
545           LOG.debug("use specific epithet as author prefix");
546           cn.setSpecificEpithet(null);
547           cn.setAuthorship(extendedAuthor);
548         }
549       }

583   /**
584    * Fully parse the supplied name also trying to extract authorships, a conceptual sec reference, remarks or notes
585    * on the nomenclatural status. In some cases the authorship parsing proves impossible and this nameparser will
586    * return null.
587    *
588    * For strings which are no scientific names and scientific names that cannot be expressed by the ParsedName class
589    * the parser will throw an UnparsableException with a given NameType and the original, unparsed name. This is the
590    * case for all virus names and proper hybrid formulas, so make sure you catch and process this exception.
591    */
592   public ParsedName parse(final String scientificName) throws UnparsableException {
593     if (Strings.isNullOrEmpty(scientificName)) {
594       throw new UnparsableException(NameType.BLACKLISTED, scientificName);
595     }
596     long start = 0;
597     if (LOG.isDebugEnabled()) {
598       start = System.currentTimeMillis();
599     }
600 
601     ParsedName pn = new ParsedName();
602     pn.setScientificName(scientificName);
603 
604     String name = scientificName;
605 
606     // before any cleaning test for properly quoted candidate names
607     Matcher m = IS_CANDIDATUS_QUOTE_PATTERN.matcher(scientificName);
608     if (m.find()) {
609       pn.setType(NameType.CANDIDATUS);
610       name = m.replaceFirst(m.group(2));
611     }
612 
613     // clean name, removing seriously wrong things
614     name = preClean(name);
615     // normalise name
616     name = normalize(name);
617     if (Strings.isNullOrEmpty(name)) {
618       throw new UnparsableException(NameType.BLACKLISTED, scientificName);
619     }
620 
621     // parse cultivar names first BEFORE we strongly normalize
622     // this will potentially remove quotes needed to find cultivar names
623     m = CULTIVAR.matcher(name);
624     if (m.find()) {
625       pn.setCultivarEpithet(m.group(1));
626       name = m.replaceFirst(" ");
627       pn.setType(NameType.CULTIVAR);
628       pn.setRank(Rank.CULTIVAR);
629       LOG.debug("Cultivar: {}", pn.getCultivarEpithet());
630     }
631 
632     // detect unparsable names
633     m = BLACKLISTED.matcher(name);
634     if (m.find()) {
635       throw new UnparsableException(NameType.BLACKLISTED, scientificName);
636     }
637     // name without any latin char letter at all?
638     m = NO_LETTERS.matcher(name);
639     if (m.find()) {
640       throw new UnparsableException(NameType.BLACKLISTED, scientificName);
641     }
642     m = HYBRID_FORMULA_PATTERN.matcher(name);
643     if (m.find()) {
644       throw new UnparsableException(NameType.HYBRID, scientificName);
645     }
646     m = IS_VIRUS_PATTERN.matcher(name);
647     if (m.find()) {
648       throw new UnparsableException(NameType.VIRUS, scientificName);
649     }
650 
651     m = IS_CANDIDATUS_PATTERN.matcher(name);
652     if (m.find()) {
653       pn.setType(NameType.CANDIDATUS);
654       name = m.replaceFirst("");
655     }
656 
657     // extract nom.illeg. and other nomen status notes
658     m = EXTRACT_NOMSTATUS.matcher(name);
659     if (m.find()) {
660       pn.setNomStatus(StringUtils.trimToNull(m.group(1)));
661       name = m.replaceFirst("");
662     }
663 
664     // extract sec reference
665     m = EXTRACT_SENSU.matcher(name);
666     if (m.find()) {
667       pn.setSensu(StringUtils.trimToNull(m.group(1)));
668       name = m.replaceFirst("");
669     }
670     // extract other remarks
671     m = EXTRACT_REMARKS.matcher(name);
672     if (m.find()) {
673       pn.setRemarks(StringUtils.trimToNull(m.group(1)));
674       name = m.replaceFirst("");
675     }
676 
677     // check for indets
678     m = RANK_MARKER_AT_END.matcher(name);
679     // f. is a marker for forms, but more often also found in authorships as "filius" - son of.
680     // so ignore those
681     if (m.find() && !(name.endsWith(" f.") || name.endsWith(" f"))) {
682       pn.setType(NameType.INFORMAL);
683       pn.setRankMarker(m.group(2));
684       name = m.replaceAll(" ");
685     }
686     m = NORM_INDET.matcher(name);
687     if (m.find()) {
688       pn.setType(NameType.INFORMAL);
689       name = m.replaceAll(" ");
690     }
691 
692     name = normalizeStrong(name);
693     if (Strings.isNullOrEmpty(name)) {
694       throw new UnparsableException(NameType.DOUBTFUL, scientificName);
695     }
696 
697     // remember current rank for later reuse
698     Rank origRank = pn.getRank();
699 
700     // try regular parsing
701     int passNo = 1;
702     boolean parsed = parseNormalisedName(pn, name);
703     if (!parsed) {
704       // try again with stronger normalisation, maybe that helps...
705       LOG.debug("Can't parse, use dirty normalizer");
706       final String deDirtedName = cleanStrong(name);
707       parsed = parseNormalisedName(pn, deDirtedName);
708       passNo++;
709       if (!parsed) {
710         LOG.debug("Still can't parse, try to ignore authors");
711         // try to parse canonical alone ignoring authorship as last resort
712         parsed = parseNormalisedNameIgnoreAuthors(pn, deDirtedName);
713         passNo++;
714         pn.setAuthorsParsed(false);
715         if (!parsed) {
716           // we just cant parse this one
717           // try to spot a virus name once we know its not a scientific name
718           m = IS_VIRUS_PATTERN_POSTFAIL.matcher(name);
719           if (m.find()) {
720             throw new UnparsableException(NameType.VIRUS, scientificName);
721           }
722 
723           throw new UnparsableException(NameType.DOUBTFUL, scientificName);
724         }
725       }
726     }
727     if (origRank != null) {
728       pn.setRank(origRank);
729     }
730 
731     // primarily make sure again we haven't parsed a non name
732     postAssertParsing(pn, scientificName, name);
733 
734     // determine name type if not yet assigned
735     if (pn.getType() == null) {
736       // a doubtful name?
737       m = DOUBTFUL.matcher(scientificName);
738       if (!m.find()) {
739         pn.setType(NameType.DOUBTFUL);
740       } else {
741         // a wellformed, code compliant name?
742         if (scientificName.equals(pn.canonicalNameComplete())) {
743           pn.setType(NameType.WELLFORMED);
744         } else {
745           pn.setType(NameType.SCINAME);
746         }
747       }
748     }
749 
750     LOG.debug("Parsing time: {}", (System.currentTimeMillis() - start));
751 
752     return pn;
753   }

757     // if we only match a monomial in the 3rd pass its suspicious

766   private boolean parseNormalisedName(ParsedName cn, String scientificName) {
767     LOG.debug("Parse normed name string: {}", scientificName);
768     Matcher matcher = NAME_PATTERN.matcher(scientificName);
769     if (matcher.find() && matcher.group(0).equals(scientificName)) {
770       if (LOG.isDebugEnabled()) {
771         logMatcher(matcher);
772       }
773       cn.setGenusOrAbove(StringUtils.trimToNull(matcher.group(1)));
774       boolean bracketSubrankFound = false;
775       if (matcher.group(2) != null) {
776         bracketSubrankFound = true;
777         cn.setInfraGeneric(StringUtils.trimToNull(matcher.group(2)));
778       } else if (matcher.group(4) != null) {
779         String rank = StringUtils.trimToNull(matcher.group(3));
780         if (!rank.endsWith(".")) {
781           rank = rank + ".";
782         }
783         cn.setRankMarker(rank);
784         cn.setInfraGeneric(StringUtils.trimToNull(matcher.group(4)));
785       }
786       cn.setSpecificEpithet(StringUtils.trimToNull(matcher.group(5)));
787       // #6 is filling authors or ranks in the middle not stored in ParsedName
788       if (matcher.group(7) != null && matcher.group(7).length() > 1) {
789         cn.setRankMarker(StringUtils.trimToNull(matcher.group(7)));
790       }
791       cn.setInfraSpecificEpithet(StringUtils.trimToNull(matcher.group(8)));
792 
793       // #9 is entire authorship, not stored in ParsedName
794       cn.setBracketAuthorship(StringUtils.trimToNull(matcher.group(10)));
795       if (bracketSubrankFound && cn.getBracketAuthorship() == null && cn.getSpecificEpithet() == null && !MONOMIALS
796         .contains(cn.getInfraGeneric())) {
797         // rather an author than a infrageneric rank. Swap
798         cn.setBracketAuthorship(cn.getInfraGeneric());
799         cn.setInfraGeneric(null);
800         LOG.debug("swapped subrank with bracket author: {}", cn.getBracketAuthorship());
801       }
802       if (matcher.group(11) != null && matcher.group(11).length() > 2) {
803         String yearAsString = matcher.group(11).trim();
804         cn.setBracketYear(yearAsString);
805       }
806       cn.setAuthorship(StringUtils.trimToNull(matcher.group(12)));
807       if (matcher.group(13) != null && matcher.group(13).length() > 2) {
808         String yearAsString = matcher.group(13).trim();
809         cn.setYear(yearAsString);
810       }
811 
812       // make sure (infra)specific epithet is not a rank marker!
813       lookForIrregularRankMarker(cn);
814       // 2 letter epitheta can also be author prefixes - check that programmatically, not in regex
815       checkEpithetVsAuthorPrefx(cn);
816       return true;
817     }
818     return false;
819   }

830     LOG.debug("Parse normed name string ignoring authors: {}", scientificName);
831

833     Matcher matcher = CANON_NAME_IGNORE_AUTHORS.matcher(scientificName);
834     boolean matchFound = matcher.find();

844           // rather an author...

847       } else if (matcher.group(4) != null) {

851         cn.setInfraGeneric(StringUtils.trimToNull(matcher.group(4)));

854       if (matcher.group(6) != null && matcher.group(6).length() > 1) {
855         cn.setRankMarker(StringUtils.trimToNull(matcher.group(6)));
856       }
857       if (matcher.group(7) != null && matcher.group(7).length() >= 2) {
858         setCanonicalInfraSpecies(cn, matcher.group(7));
859       } else {
860         setCanonicalInfraSpecies(cn, matcher.group(8));
861       }
862 
863       // make sure (infra)specific epithet is not a rank marker!
864       lookForIrregularRankMarker(cn);
865 
866       return true;
867     }
