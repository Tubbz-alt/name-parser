27  /**
28   * Core parser class of the name parser that tries to take a clean name into its pieces by using regular expressions.
29   * It runs the actual regex matching in another thread that stops whenever the configured timeout is reached.
30   */
31  public class NormalisedNameParser {

41    protected static final String NAME_LETTERS = "A-ZÏËÖÜÄÉÈČÁÀÆŒ";
42    protected static final String name_letters = "a-zïëöüäåéèčáàæœ";
43    protected static final String AUTHOR_LETTERS = NAME_LETTERS + "\\p{Lu}"; // upper case unicode letter, not numerical
44    // (\W is alphanum)
45    protected static final String author_letters = name_letters + "\\p{Ll}"; // lower case unicode letter, not numerical
46    // (\W is alphanum)
47    protected static final String AUTHOR_PREFIXES =
48      "(?:[vV](?:an)(?:[ -](?:den|der) )? ?|von[ -](?:den |der |dem )?|(?:del|Des|De|de|di|Di|da|N)[`' _]|(?:de )?(?:la|le) |d'|D'|Mac|Mc|Le|St\\.? ?|Ou|O')";
49    protected static final String AUTHOR = "(?:" +
50                                           // author initials
51                                           "(?:" + "(?:[" + AUTHOR_LETTERS + "]{1,3}\\.?[ -]?){0,3}" +
52                                           // or full first name
53                                           "|[" + AUTHOR_LETTERS + "][" + author_letters + "?]{3,}" + " )?" +
54                                           // common prefixes
55                                           AUTHOR_PREFIXES + "?" +
56                                           // only allow v. in front of Capital Authornames - if included in AUTHOR_PREFIXES parseIgnoreAuthors fails
57                                           "(?:v\\. )?" +
58                                           // regular author name
59                                           "[" + AUTHOR_LETTERS + "]+[" + author_letters + "?]*\\.?" +
60                                           // potential double names, e.g. Solms-Laub.
61                                           "(?:(?:[- ](?:de|da|du)?[- ]?)[" + AUTHOR_LETTERS + "]+[" + author_letters
62                                           + "?]*\\.?)?" +
63                                           // common name suffices (ms=manuscript, not yet published)
64                                           "(?: ?(?:f|fil|j|jr|jun|junior|sr|sen|senior|ms)\\.?)?" +
65                                           // at last there might be 2 well known sanction authors for fungus, see POR-2454
66                                           "(?: *: *(?:Pers|Fr)\\.?)?" +
67                                           ")";
68    protected static final String AUTHOR_TEAM =
69      AUTHOR + "?(?:(?: ?ex\\.? | & | et | in |, ?|; ?|\\.)(?:" + AUTHOR + "|al\\.?))*";
70    protected static final Pattern AUTHOR_TEAM_PATTERN = Pattern.compile("^" + AUTHOR_TEAM + "$");
71    protected static final String YEAR = "[12][0-9][0-9][0-9?][abcdh?]?(?:[/-][0-9]{1,4})?";
72    // protected static final String YEAR_RANGE = YEAR+"(?: ?-? ?"+YEAR+")?";

74    protected static final String RANK_MARKER_SPECIES =
75      "(?:notho)?(?:" + StringUtils.join(Rank.RANK_MARKER_MAP_INFRASPECIFIC.keySet(), "|") + "|agg)\\.?";
76  
77    protected static final String RANK_MARKER_MICROBIAL =
78      "(?:bv\\.|ct\\.|f\\. ?sp\\.|" + StringUtils.join(Lists.transform(Lists.newArrayList(Rank.INFRASUBSPECIFIC_MICROBIAL_RANKS), new Function<Rank, String>() {
79                  @Override
80                  public String apply(Rank rank) {
81                      return rank.getMarker().replaceAll("\\.", "\\\\.");
82                  }
83              }
84      ), "|") + ")";

86    protected static final String EPHITHET_PREFIXES = "van|novae";
87    protected static final String EPHITHET_UNALLOWED_ENDINGS = "\\bex|var|type|form";
88    protected static final String EPHITHET = "(?:[0-9]+-|[doml]')?"
89              + "(?:(?:" + EPHITHET_PREFIXES + ") [a-z])?"
90              + "[" + name_letters + "+-]{1,}(?<! d)[" + name_letters + "]"
91              // avoid epithets ending with the unallowed endings, e.g. serovar
92              + "(?<!(?:"+EPHITHET_UNALLOWED_ENDINGS+"))(?=\\b)";
93    protected static final String MONOMIAL =
94      "[" + NAME_LETTERS + "](?:\\.|[" + name_letters + "]+)(?:-[" + NAME_LETTERS + "]?[" + name_letters + "]+)?";
95    // a pattern matching typical latin word endings. Helps identify name parts from authors
96    private static final Pattern LATIN_ENDINGS;
97    static {
98        try {
99            List<String> endings = FileUtils.streamToList(FileUtils.classpathStream("latin-endings.txt"));
100           LATIN_ENDINGS = Pattern.compile("(" + Joiner.on('|').skipNulls().join(endings) + ")$");
101       } catch (IOException e) {
102           throw new IllegalStateException("Failed to read latin-endings.txt from classpath resources", e);
103       }
104   }
105   protected static final String INFRAGENERIC =
106     "(?:" + "\\( ?([" + NAME_LETTERS + "][" + name_letters + "-]+) ?\\)" + "|" + "(" + StringUtils
107       .join(Rank.RANK_MARKER_MAP_INFRAGENERIC.keySet(), "|") + ")\\.? ?([" + NAME_LETTERS + "][" + name_letters + "-]+)"
108     + ")";

110   protected static final String RANK_MARKER_ALL = "(notho)? *(" + StringUtils.join(Rank.RANK_MARKER_MAP.keySet(), "|") + ")\\.?";
111   private static final Pattern RANK_MARKER_ONLY = Pattern.compile("^" + RANK_MARKER_ALL + "$");

114   public static final Pattern CANON_NAME_IGNORE_AUTHORS = Pattern.compile("^" +
115               // #1 genus/monomial
116               "(×?" + MONOMIAL + ")" +
117               // #2 or #4 subgenus/section with #3 infrageneric rank marker
118               "(?:(?<!ceae) " + INFRAGENERIC + ")?" +
119               // catch author name prefixes just to ignore them so they dont become wrong epithets
120               "(?: " + AUTHOR_PREFIXES + ")?" +
121               // #5 species
122               "(?: (×?" + EPHITHET + "))?" +
123               // catch author name prefixes just to ignore them so they dont become wrong epithets
124               "(?: " + AUTHOR_PREFIXES + ")?" +
125               "(?:" +
126                 // either directly a infraspecific epitheton or a author but then mandate rank marker
127                 "(?:" +
128                   // anything in between
129                   ".*" +
130                   // #6 infraspecies rank
131                   "( " + RANK_MARKER_SPECIES + "[ .])" +
132                   // #7 infraspecies epithet
133                   "(×?" + EPHITHET + ")" +
134                 ")|" +
135                   // #8 infraspecies epithet
136                 " (×?" + EPHITHET + ")" +
137               ")?" +
138               "(?: " +
139                 // #9 microbial rank
140                 "(" + RANK_MARKER_MICROBIAL + ")[ .]" +
141                 // #10 microbial infrasubspecific epithet
142                 "(\\S+)" +
143               ")?");

145   public static final Pattern NAME_PATTERN = Pattern.compile("^" +
146              // #1 genus/monomial
147              "(×?" + MONOMIAL + ")" +
148              // #2 or #4 subgenus/section with #3 infrageneric rank marker
149              "(?:(?<!ceae) " + INFRAGENERIC + ")?" +
150              // #5 species
151              "(?: (×?" + EPHITHET + "))?" +
152 
153              "(?:" +
154                "(?:" +
155                  // #6 strip out intermediate, irrelevant authors or infraspecific ranks in case of quadrinomials
156                  "( .*?)?" +
157                  // #7 infraspecies rank
158                  "( " + RANK_MARKER_SPECIES + ")" +
159                ")?" +
160                // #8 infraspecies epitheton
161                "(?: (×?\"?" + EPHITHET + "\"?))" +
162              ")?" +
163 
164              "(?: " +
165                // #9 microbial rank
166                "(" + RANK_MARKER_MICROBIAL + ")[ .]" +
167                // #10 microbial infrasubspecific epithet
168                "(\\S+)" +
169              ")?" +
170 
171              // #11 entire authorship incl basionyms and year
172              "(,?" +
173                "(?: ?\\(" +
174                  // #12 basionym authors
175                  "(" + AUTHOR_TEAM + ")?" +
176                  // #13 basionym year
177                  ",?( ?" + YEAR + ")?" +
178                "\\))?" +
179 
180                // #14 authors
181                "( " + AUTHOR_TEAM + ")?" +
182                // #15 year with or without brackets
183                "(?: ?\\(?,? ?(" + YEAR + ")\\)?)?" +
184              ")" +
185              "$");

188   private class MatcherCallable implements Callable<Matcher> {
189     private final String scientificName;
190 
191     MatcherCallable(String scientificName) {
192       this.scientificName = scientificName;
193     }
194 
195     @Override
196     public Matcher call() throws Exception {
197       Matcher matcher = NAME_PATTERN.matcher(scientificName);
198       matcher.find();
199       return matcher;
200     }
201   }

203   /**
204    * Tries to parse a name string with the full regular expression.
205    * In very few, extreme cases names with very long authorships might cause the regex to never finish or take hours
206    * we run this parsing in a separate thread that can be stopped if it runs too long.
207    * @param cn
208    * @param scientificName
209    * @param rank the rank of the name if it is known externally. Helps identifying infrageneric names vs bracket authors
210    * @return
211    */
212   public boolean parseNormalisedName(ParsedName cn, String scientificName, @Nullable Rank rank) {
213     LOG.debug("Parse normed name string: {}", scientificName);
214     FutureTask<Matcher> task = new FutureTask<Matcher>(new MatcherCallable(scientificName));
215     THREAD_POOL.execute(task);
216 
217     try {
218       Matcher matcher = task.get(timeout, TimeUnit.MILLISECONDS);
219       // if there was no match in the callable, this can yield an IllegalStateException
220       if (matcher.group(0).equals(scientificName)) {
221         if (LOG.isDebugEnabled()) {
222           logMatcher(matcher);
223         }
224         cn.setGenusOrAbove(StringUtils.trimToNull(matcher.group(1)));
225         boolean bracketSubrankFound = false;
226         if (matcher.group(2) != null) {
227           bracketSubrankFound = true;
228           cn.setInfraGeneric(StringUtils.trimToNull(matcher.group(2)));
229         } else if (matcher.group(4) != null) {
230           String rankMarker = StringUtils.trimToNull(matcher.group(3));
231           if (!rankMarker.endsWith(".")) {
232               rankMarker = rankMarker + ".";
233           }
234           cn.setRankMarker(rankMarker);
235           cn.setInfraGeneric(StringUtils.trimToNull(matcher.group(4)));
236         }
237         cn.setSpecificEpithet(StringUtils.trimToNull(matcher.group(5)));
238         // #6 is filling authors or ranks in the middle not stored in ParsedName
239         if (matcher.group(7) != null && matcher.group(7).length() > 1) {
240           cn.setRankMarker(StringUtils.trimToNull(matcher.group(7)));
241         }
242         cn.setInfraSpecificEpithet(StringUtils.trimToNull(matcher.group(8)));
243 
244         // microbial ranks
245         if (matcher.group(9) != null) {
246           cn.setRankMarker(matcher.group(9));
247           cn.setInfraSpecificEpithet(matcher.group(10));
248         }
249 
250         // #11 is entire authorship, not stored in ParsedName
251         cn.setBracketAuthorship(StringUtils.trimToNull(matcher.group(12)));
252         if (bracketSubrankFound && infragenericIsAuthor(cn, rank)) {
253           // rather an author than a infrageneric rank. Swap
254           cn.setBracketAuthorship(cn.getInfraGeneric());
255           cn.setInfraGeneric(null);
256           LOG.debug("swapped subrank with bracket author: {}", cn.getBracketAuthorship());
257         }
258         if (matcher.group(13) != null && matcher.group(13).length() > 2) {
259           String yearAsString = matcher.group(13).trim();
260           cn.setBracketYear(yearAsString);
261         }
262         cn.setAuthorship(StringUtils.trimToNull(matcher.group(14)));
263         if (matcher.group(15) != null && matcher.group(15).length() > 2) {
264           String yearAsString = matcher.group(15).trim();
265           cn.setYear(yearAsString);
266         }
267 
268         // make sure (infra)specific epithet is not a rank marker!
269         lookForIrregularRankMarker(cn);
270         // 2 letter epitheta can also be author prefixes - check that programmatically, not in regex
271         checkEpithetVsAuthorPrefx(cn);
272 
273         // if no rank was parsed but given externally use it!
274         if (cn.getRankMarker() == null && rank != null) {
275             cn.setRank(rank);
276         }
277         return true;
278       }
279 
280     } catch (InterruptedException e) {
281       LOG.warn("InterruptedException for name: {}", scientificName, e);
282     } catch (ExecutionException e) {
283       LOG.warn("ExecutionException for name: {}", scientificName, e);
284     } catch (IllegalStateException e) {
285       // we simply had no match
286     } catch (TimeoutException e) {
287       // timeout
288       LOG.info("Parsing timeout for name: {}", scientificName);
289     }
290 
291     return false;
292   }

294     private boolean infragenericIsAuthor(ParsedName pn, Rank rank) {
295         return pn.getBracketAuthorship() == null && pn.getSpecificEpithet() == null && (
296                 rank != null && !(rank.isInfrageneric() && !rank.isSpeciesOrBelow())
297                         //|| pn.getInfraGeneric().contains(" ")
298                         || rank == null && !LATIN_ENDINGS.matcher(pn.getInfraGeneric()).find()
299         );
300     }

302   /**
303    *
304    * @param cn
305    * @param scientificName
306    * @param rank the rank of the name if it is known externally. Helps identifying infrageneric names vs bracket authors
307    * @return
308    */
309   public boolean parseNormalisedNameIgnoreAuthors(ParsedName cn, String scientificName, @Nullable Rank rank) {
310     LOG.debug("Parse normed name string ignoring authors: {}", scientificName);
311 
312     // match for canonical
313     Matcher matcher = CANON_NAME_IGNORE_AUTHORS.matcher(scientificName);
314     boolean matchFound = matcher.find();
315     if (matchFound) {
316       if (LOG.isDebugEnabled()) {
317         logMatcher(matcher);
318       }
319       cn.setGenusOrAbove(StringUtils.trimToNull(matcher.group(1)));
320       if (matcher.group(2) != null) {
321         // subrank in paranthesis. Not an author?
322         cn.setInfraGeneric(StringUtils.trimToNull(matcher.group(2)));
323         if (infragenericIsAuthor(cn, rank)) {
324             // rather an author...
325             cn.setInfraGeneric(null);
326         }
327       } else if (matcher.group(4) != null) {
328         // infrageneric with rank indicator given
329         String rankMarker = StringUtils.trimToNull(matcher.group(3));
330         cn.setRankMarker(rankMarker);
331         cn.setInfraGeneric(StringUtils.trimToNull(matcher.group(4)));
332       }
333       cn.setSpecificEpithet(StringUtils.trimToNull(matcher.group(5)));
334       if (matcher.group(6) != null && matcher.group(6).length() > 1) {
335         cn.setRankMarker(matcher.group(6));
336       }
337       if (matcher.group(7) != null && matcher.group(7).length() >= 2) {
338         setCanonicalInfraSpecies(cn, matcher.group(7));
339       } else {
340         setCanonicalInfraSpecies(cn, matcher.group(8));
341       }
342       if (matcher.group(9) != null) {
343         cn.setRankMarker(matcher.group(9));
344         cn.setInfraSpecificEpithet(matcher.group(10));
345       }
346 
347       // make sure (infra)specific epithet is not a rank marker!
348       lookForIrregularRankMarker(cn);
349 
350       return true;
351     }
352     return false;
353   }

355   private static void setCanonicalInfraSpecies(ParsedName pn, String epi) {
356     if (epi == null || epi.equalsIgnoreCase("sec") || epi.equalsIgnoreCase("sensu")) {
357       return;
358     }
359     pn.setInfraSpecificEpithet(StringUtils.trimToNull(epi));
360   }

362   /**
363    * if no rank marker is set yet inspect epitheta for wrongly placed rank markers and modify parsed name accordingly.
364    * This is sometimes the case for informal names like: Coccyzus americanus ssp.
365    *
366    * @param cn the already parsed name
367    */
368   private void lookForIrregularRankMarker(ParsedName cn) {
369     if (cn.getRankMarker() == null) {
370       if (cn.getInfraSpecificEpithet() != null) {
371         Matcher m = RANK_MARKER_ONLY.matcher(cn.getInfraSpecificEpithet());
372           if (m.find()) {
373               // we found a rank marker, make it one
374           cn.setRankMarker(cn.getInfraSpecificEpithet());
375           cn.setInfraSpecificEpithet(null);
376         }
377       } else if (cn.getSpecificEpithet() != null) {
378         Matcher m = RANK_MARKER_ONLY.matcher(cn.getSpecificEpithet());
379         if (m.find()) {
380           // we found a rank marker, make it one
381           cn.setRankMarker(cn.getSpecificEpithet());
382           cn.setSpecificEpithet(null);
383         }
384       }
385     }
386   }

388   /**
389    * 2 letter epitheta can also be author prefixes - check that programmatically, not in regex
390    */
391   private void checkEpithetVsAuthorPrefx(ParsedName cn) {
392     if (cn.getRankMarker() == null) {
393       if (cn.getInfraSpecificEpithet() != null) {
394         // might be subspecies without rank marker
395         // or short authorship prefix in epithet. test
396         String extendedAuthor = cn.getInfraSpecificEpithet() + " " + cn.getAuthorship();
397         Matcher m = AUTHOR_TEAM_PATTERN.matcher(extendedAuthor);
398         if (m.find()) {
399           // matches author. Prefer that
400           LOG.debug("use infraspecific epithet as author prefix");
401           cn.setInfraSpecificEpithet(null);
402           cn.setAuthorship(extendedAuthor);
403         }
404       } else {
405         // might be monomial with the author prefix erroneously taken as the species epithet
406         String extendedAuthor = cn.getSpecificEpithet() + " " + cn.getAuthorship();
407         Matcher m = AUTHOR_TEAM_PATTERN.matcher(extendedAuthor);
408         if (m.find()) {
409           // matches author. Prefer that
410           LOG.debug("use specific epithet as author prefix");
411           cn.setSpecificEpithet(null);
412           cn.setAuthorship(extendedAuthor);
413         }
414       }
415     }
416   }

418   private void logMatcher(Matcher matcher) {
419     int i = -1;
420     while (i < matcher.groupCount()) {
421       i++;
422       LOG.debug("  {}: >{}<", i, matcher.group(i));
423     }
